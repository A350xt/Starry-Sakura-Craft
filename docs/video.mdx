import React, { useState } from 'react';

export const VideoCard = ({ title, coverImage, link, description, videoTitle }) => {
  const [isHovered, setIsHovered] = useState(false);

  return (
    <div 
      style={{
        backgroundColor: 'white',
        borderRadius: '8px',
        boxShadow: isHovered ? '0 8px 15px rgba(0, 0, 0, 0.15)' : '0 4px 6px rgba(0, 0, 0, 0.1)',
        overflow: 'hidden',
        border: '1px solid #e5e7eb',
        padding: '16px',
        transform: isHovered ? 'scale(1.02)' : 'scale(1)',
        transition: 'all 0.2s ease-in-out',
        cursor: 'pointer'
      }}
      onMouseEnter={() => setIsHovered(true)}
      onMouseLeave={() => setIsHovered(false)}
    >
      <h3 style={{ 
        fontSize: '18px', 
        fontWeight: '600', 
        marginBottom: '12px',
        marginTop: '0'
      }}>
        {title}
      </h3>
      <div style={{
        width: 'auto',
        height: 'auto',
        marginBottom: '12px',
        display: 'flex',
        justifyContent: 'center',
        alignItems: 'center',
        backgroundColor: '#f9fafb',
        borderRadius: '6px'
      }}>
        <img 
          src={coverImage} 
          alt={title}
          style={{
            maxWidth: '100%',
            maxHeight: '100%',
            width: 'auto',
            height: 'auto',
            borderRadius: '6px'
          }}
        />
      </div>
      <a 
        href={link}
        target="_blank"
        rel="noopener noreferrer"
        style={{
          color: '#2563eb',
          fontWeight: '500',
          textDecoration: 'none',
          display: 'block',
          marginBottom: '12px',
          whiteSpace: 'pre-line'
        }}
      >
        {videoTitle}
      </a>
      <p style={{ 
        color: '#6b7280', 
        fontSize: '14px',
        margin: '0',
        whiteSpace: 'pre-line'
      }}>
        {description}
      </p>
    </div>
  );
};
